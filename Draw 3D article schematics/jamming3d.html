clc
clear

[X,Y] = meshgrid(-60000:5000:60000);%大面积用-60000:5000:60000，小面积用-20000:5000:20000
Z = X.^2-2*Y.^2;
Z=Z/160000;

x0=[];y0=[];z0=[];
px=[];py=[];pz=[];

% 设置无限远平行光源光照效果
figure
surf(X,Y,Z);
light('Style','infinit','Position',[0 -0.6 1]);
axis equal;
%以上是绘制曲面

[x,y,z] = sphere;
CO(:,:,1) = zeros(21); % red
CO(:,:,2) = ones(21); % green
CO(:,:,3) = ones(21); % blue

hold on
for i=1:20000
    x0(i)=-60000+120000*rand; %大面积用-60000+120000*rand;，小面积用-55000+110000*rand
    y0(i)=-60000+120000*rand;
    z0(i)= (x0(i).^2-2*y0(i).^2)/160000;
    overlap=0;
      for j=1:length(px)
        if ((px(j)-x0(i))^2+(py(j)-y0(i))^2+(pz(j)-z0(i))^2)<80000000 %为了可以jamming这里做了修改，让粒子可以重叠一部分
          overlap=1;
        end
      end
    if overlap==0
       surf(5000*x+x0(i),5000*y+y0(i),5000*z+z0(i),CO);% sphere centered at (x0(i),y0(i),z0(i))
       px=[px x0(i)];py=[py y0(i)];pz=[pz z0(i)];

    end
    if length(px)==200 %调节密集程度
        break %.结束当前整个循环，执行当前循环下边的语句。忽略循环体中任何其它语句和循环条件测试。
    end
end
shading interp %去除网格线

axis off
hold off
view([-45,40])
